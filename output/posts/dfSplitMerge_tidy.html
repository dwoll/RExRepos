Install required packages

dplyr

wants <- c("dplyr")
has   <- wants %in% rownames(installed.packages())
if(any(!has)) install.packages(wants[!has])

Simulate data

set.seed(123)
N      <- 12
sex    <- factor(sample(c("f", "m"), N, replace=TRUE), levels=c("f", "m"))
group  <- factor(sample(rep(c("CG", "WL", "T"), 4), N, replace=FALSE), levels=c("CG", "WL", "T"))
age    <- sample(18:35, N, replace=TRUE)
IQ     <- round(rnorm(N, mean=100, sd=15))
rating <- round(runif(N, min=0, max=6))
(myDf1 <- data.frame(id=1:N, sex, group, age, IQ, rating))

(myDf2 <- data.frame(id=21,
                     sex=factor("f", levels=c("f", "m")),
                     age=48,
                     rating=3))

Combine data frames

Bind rows - does not require identical order of variables / identical variables

library(dplyr)
bind_rows(myDf1, myDf2) %>%
    tail(n=3)
   id sex group age IQ rating
11 11   m    CG  32 91      4
12 12   m    WL  27 98      2
13 21   f  <NA>  48 NA      3

Left outer join

(IDDV <- data.frame(ID=factor(rep(1:3, each=2)),
                    DV=round(rnorm(6, 100, 15))))
  ID  DV
1  1  82
2  1 107
3  2 112
4  2  98
5  3 119
6  3  86
(IV <- data.frame(ID=factor(1:3),
                  IV=factor(c("A", "B", "A")),
                  sex=factor(c("f", "f", "m"))))
  ID IV sex
1  1  A   f
2  2  B   f
3  3  A   m
IDDV %>%
    left_join(IV, by="ID")
  ID  DV IV sex

2  1 107  A   f
3  2 112  B   f
4  2  98  B   f
5  3 119  A   m
6  3  86  A   m

Specify identical variables which have different names in the two data sets.

(dfA <- data.frame(ID=1:4,
                   initials=c("AB", "CD", "EF", "GH"),
                   IV1=c("-", "-", "+", "+"),
                   DV1=c(10, 10, 11, 14),
                   stringsAsFactors=FALSE))
  ID initials IV1 DV1
1  1       AB   -  10
2  2       CD   -  10
3  3       EF   +  11
4  4       GH   +  14
(dfB <- data.frame(ID_mod=3:6,
                   initials=c("EF", "GH", "IJ", "KL"),
                   IV2=c("A", "B", "A", "B"),
                   DV2=c(92, 79, 101, 81),
                   stringsAsFactors=FALSE))
  ID_mod initials IV2 DV2
1      3       EF   A  92
2      4       GH   B  79
3      5       IJ   A 101
4      6       KL   B  81
dfA %>%
    left_join(dfB, by=c("ID"="ID_mod", "initials"))
  ID initials IV1 DV1  IV2 DV2
1  1       AB   -  10 <NA>  NA
2  2       CD   -  10 <NA>  NA
3  3       EF   +  11    A  92
4  4       GH   +  14    B  79
